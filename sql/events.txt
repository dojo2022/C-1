//eventsテーブル作成
CREATE TABLE events (
number INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
event VARCHAR(20) NOT NULL,
type  INT NOT NULL,
level INT NOT NULL,
available INT NOT NULL DEFAULT 0,
user_id VARCHAR(20) NOT NULL
);
